    const compressImage = (file, maxResolution = 800) => {
      return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onload = (event) => {
          const img = new Image();
          img.onload = () => {
            const canvas = document.createElement('canvas');
            let { width, height } = img;
            if (width > height && width > maxResolution) { height *= maxResolution / width; width = maxResolution; } 
            else if (height > maxResolution) { width *= maxResolution / height; height = maxResolution; }
            canvas.width = width; canvas.height = height;
            const ctx = canvas.getContext('2d');
            ctx.drawImage(img, 0, 0, width, height);
            resolve(canvas.toDataURL('image/jpeg', maxResolution > 1000 ? 0.8 : 0.6));
          };
          img.src = event.target.result;
        };
        reader.readAsDataURL(file);
      });
    };

                    {isStockTaking ? '⏳ AI 批次辨識中...' : '📸 點擊拍照進行盤點'}
                  </button>
                  {inventory.length === 0 && <p className="text-red-500 text-xs mt-3 font-bold">⚠️ 庫存為空，請先建檔</p>}
                </div>
                <div className="bg-white p-6 rounded-3xl shadow-sm border border-gray-100">
                  <div className="flex justify-between items-center mb-4"><h3 className="font-black text-gray-900">💾 盤點暫存</h3><span className="bg-blue-50 text-blue-700 px-3 py-1 rounded-full text-xs font-bold">已掃描 {Object.keys(scanSession).reduce((acc, itemId) => acc + Object.values(scanSession[itemId]).reduce((sum, val) => sum + val, 0), 0)} 件</span></div>
                  <div className="bg-gray-50 rounded-2xl p-4 h-40 overflow-y-auto mb-4 border border-gray-100">
                    {scanLogs.length === 0 ? <div className="text-center text-sm text-gray-400 h-full flex items-center justify-center">拍照後，結果將暫存於此</div> : (
                      <div className="space-y-2">{scanLogs.map(log => (<div key={log.id} className={`text-sm flex items-center p-2.5 rounded-xl border ${log.type === 'error' ? 'bg-red-50 border-red-100' : 'bg-white border-gray-100'}`}><span className="font-bold text-gray-700">{log.text}</span></div>))}</div>
                    )}
                  </div>
                  <button onClick={() => setShowCompareModal(true)} disabled={Object.keys(scanSession).length === 0} className="w-full bg-gray-900 text-white py-4 rounded-xl font-bold disabled:bg-gray-200 shadow-md">結束盤點並對帳 ➡️</button>
                </div>
              </div>
            )}

            {activeTab === 'upload' && (
              <div>
                {isScanning ? (
                  <div className="bg-white rounded-3xl p-20 text-center shadow-sm border border-gray-100"><div className="text-6xl mb-4 animate-bounce">🧠</div><h2 className="text-xl font-black">AI 視覺萃取中...</h2></div>
                ) : drafts.length === 0 ? (
                  <div onClick={() => fileInputRef.current.click()} className="bg-white rounded-3xl border-2 border-dashed border-gray-200 hover:border-gray-900 p-12 md:p-20 text-center cursor-pointer">
                    <div className="text-6xl mb-4">🔲</div>
                    <h2 className="text-2xl md:text-3xl font-black text-gray-900 mb-3">拍照批次建檔</h2>
                    <p className="text-gray-500 mb-8 text-sm">拍一張包含單副或 3x3 的照片。AI 將自動為您建檔。</p>
                    <div className="inline-flex items-center gap-2 bg-gray-900 text-white px-8 py-4 rounded-xl font-bold shadow-lg">📸 開啟相機拍照</div>
                  </div>
                ) : (
                  <div className="space-y-6">
                    <div className="flex justify-between items-center bg-gray-900 text-white p-4 rounded-2xl shadow-lg">
                      <div><h2 className="font-black">✅ 確認建檔清單</h2><p className="text-xs text-gray-300">AI 辨識出 {drafts.length} 副。</p></div>
                      <button onClick={() => {setDrafts([]); setUploadedMediaRaw(null);}} className="text-sm font-bold bg-white/20 px-3 py-1.5 rounded-lg">重拍</button>
                    </div>
                    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
                      {drafts.map(draft => (
                        <div key={draft.uid} className={`bg-white rounded-3xl p-5 shadow-sm border-2 relative ${draft.isMatch ? 'border-blue-200' : 'border-gray-100'}`}>
                          <button onClick={() => removeDraft(draft.uid)} className="absolute top-4 right-4 text-xl hover:text-red-500">❌</button>
                          <div className="mb-4">
                            {draft.isMatch ? <span className="inline-flex bg-blue-50 text-blue-700 px-2 py-1 rounded text-xs font-bold mb-2">✅ 現有款式更新</span> : <span className="inline-flex bg-pink-50 text-pink-600 px-2 py-1 rounded text-xs font-bold mb-2">➕ 新款式建檔</span>}
                            <input type="text" value={draft.formData.name} onChange={(e) => updateDraft(draft.uid, 'name', e.target.value)} className="w-full text-lg font-black border-b border-gray-200 pb-1 outline-none" placeholder="輸入名稱..." />
                          </div>
                          <div className="space-y-4">
                            <div className="flex gap-2">
                              <div className="flex-1"><label className="block text-[10px] font-bold text-gray-400 mb-1">甲型</label><select value={draft.formData.shape} onChange={(e) => updateDraft(draft.uid, 'shape', e.target.value)} className="w-full text-xs font-bold bg-gray-50 border-none rounded-lg p-2 outline-none"><option value="">選擇甲型</option>{NAIL_SHAPES.map(s => <option key={s} value={s}>{s}</option>)}</select></div>
                              <div className="flex-1"><label className="block text-[10px] font-bold text-gray-400 mb-1">標籤</label><input type="text" value={draft.formData.hashtags} onChange={(e) => updateDraft(draft.uid, 'hashtags', e.target.value)} className="w-full text-xs font-bold text-blue-600 bg-gray-50 border-none rounded-lg p-2 outline-none" placeholder="#標籤" /></div>
                            </div>
                            <div>
                              <label className="block text-[10px] font-bold text-gray-400 mb-1">偵測尺寸與庫存 (本次增加)</label>
                              <div className="flex items-center gap-2 bg-gray-50 p-2 rounded-xl">
                                 <div className="w-10 h-10 bg-white rounded-lg flex items-center justify-center font-black text-gray-900 border border-gray-100">{draft.formData.detectedSize || '?'}</div>
                                 <span className="text-gray-400 font-bold">+</span>
                                 <input type="number" min="0" value={draft.formData.stock[draft.formData.detectedSize] || 1} onChange={(e) => { const size = draft.formData.detectedSize; if(size) updateDraft(draft.uid, 'stock', { ...draft.formData.stock, [size]: parseInt(e.target.value)||0 }) }} className="w-16 text-center font-black py-1.5 rounded-lg border border-gray-200 outline-none" />
                                 <span className="text-xs text-gray-500 font-bold">副</span>
                              </div>
                            </div>
                          </div>
                        </div>
                      ))}
                    </div>
                    <button onClick={handleSaveAllDrafts} className="w-full bg-gray-900 text-white py-4 rounded-2xl font-black text-lg hover:bg-gray-800 shadow-xl">確認並全部儲存</button>
                  </div>
                )}
              </div>
            )}

            {activeTab === 'export' && (
               <div className="max-w-xl mx-auto bg-white p-10 rounded-3xl shadow-sm text-center">
                  <div className="text-6xl mb-6">📊</div><h2 className="text-2xl font-black mb-6">匯出庫存清單</h2>
                  <button onClick={exportToCSV} className="bg-gray-900 text-white px-8 py-4 rounded-xl font-bold w-full">⬇️ 下載 CSV</button>
               </div>
            )}
          </main>

          {/* ================= ✏️ 總覽編輯視窗 ================= */}
          {editingItem && (
            <div className="fixed inset-0 z-[80] bg-gray-900/80 backdrop-blur-sm flex items-center justify-center p-4">
               <div className="bg-white rounded-3xl p-6 w-full max-w-2xl max-h-[90vh] overflow-y-auto">
                  <div className="flex justify-between items-center mb-6 border-b pb-4"><h2 className="text-xl font-black">✏️ 編輯款式檔案</h2><button onClick={closeEditModal} className="text-2xl">❌</button></div>
                  <div className="space-y-6">
                     <div>
                        <div className="flex justify-between items-center mb-3"><label className="text-sm font-bold text-gray-700">商品照片</label><button onClick={() => editFileInputRef.current.click()} className="text-xs font-bold text-white bg-gray-900 px-3 py-1.5 rounded-lg">➕ 新增檔案</button><input type="file" multiple accept="image/*,video/mp4" ref={editFileInputRef} onChange={handleEditMediaUpload} className="hidden" /></div>
                        <div className="grid grid-cols-4 md:grid-cols-5 gap-3">
                          {editingItem.media?.map((m, idx) => (
                            <div key={idx} className="aspect-square rounded-xl overflow-hidden border border-gray-200 relative group bg-gray-50">
                              {m.type === 'image' ? <img src={m.url} className="w-full h-full object-cover" /> : <div className="w-full h-full flex justify-center items-center">▶️</div>}
                              <button onClick={() => removeEditMedia(idx)} className="absolute top-1 right-1 bg-red-500 text-white px-1.5 py-0.5 rounded-md text-xs">🗑️</button>
                            </div>
                          ))}
                        </div>
                     </div>
                     <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                       <div><label className="block text-xs font-bold text-gray-500 mb-1">名稱</label><input type="text" value={editingItem.name} onChange={(e) => setEditingItem({...editingItem, name: e.target.value})} className="w-full px-4 py-3 bg-gray-50 rounded-xl font-bold outline-none" /></div>
                       <div><label className="block text-xs font-bold text-gray-500 mb-1">甲型</label><select value={editingItem.shape} onChange={(e) => setEditingItem({...editingItem, shape: e.target.value})} className="w-full px-4 py-3 bg-pink-50 text-pink-700 rounded-xl font-bold outline-none"><option value="">選擇甲型...</option>{NAIL_SHAPES.map(s => <option key={s} value={s}>{s}</option>)}</select></div>
                     </div>
                     <div><label className="block text-xs font-bold text-gray-500 mb-1">自動標註</label><input type="text" value={editingItem.hashtags} onChange={(e) => setEditingItem({...editingItem, hashtags: e.target.value})} className="w-full px-4 py-3 bg-gray-50 rounded-xl font-bold text-blue-600 outline-none" /></div>
                     <div>
                       <label className="block text-xs font-bold text-gray-500 mb-2">庫存數量</label>
                       <div className="grid grid-cols-4 gap-3 bg-gray-50 p-3 rounded-xl border border-gray-100">
                         {['XS', 'S', 'M', 'L'].map(size => (<div key={size} className="text-center"><div className="text-xs mb-1 font-bold text-gray-500">{size}</div><input type="number" value={editingItem.stock[size]||0} onChange={e => setEditingItem({...editingItem, stock: {...editingItem.stock, [size]: parseInt(e.target.value)||0}})} className="w-full text-center py-2 border border-gray-200 rounded-lg font-bold outline-none" /></div>))}
                       </div>
                     </div>
                     <div className="flex justify-between items-center pt-4 border-t border-gray-100">
                        <button onClick={() => deleteItem(editingItem.id)} className="text-red-500 font-bold text-sm">🗑️ 刪除</button>
                        <button onClick={saveEditedItem} className="px-8 py-3 bg-gray-900 text-white rounded-xl font-bold">儲存修改</button>
                     </div>
                  </div>
               </div>
            </div>
          )}

          {/* ================= 📊 盤點比對視窗 ================= */}
          {showCompareModal && (
            <div className="fixed inset-0 z-[60] bg-gray-900/90 backdrop-blur-md flex items-center justify-center p-4">
              <div className="bg-white rounded-3xl shadow-2xl w-full max-w-3xl max-h-[90vh] flex flex-col">
                <div className="p-6 border-b border-gray-100 bg-gray-50 rounded-t-3xl flex justify-between items-center"><div><h2 className="text-2xl font-black text-gray-900">盤點對帳單</h2></div><button onClick={() => setShowCompareModal(false)} className="text-2xl">❌</button></div>
                <div className="flex-1 overflow-y-auto p-6">
                  <table className="w-full text-left bg-white border border-gray-200 rounded-2xl overflow-hidden">
                    <thead className="bg-gray-50 text-gray-500 text-xs uppercase font-bold"><tr><th className="px-4 py-3">商品</th><th className="px-4 py-3 text-center">尺寸</th><th className="px-4 py-3 text-center">系統</th><th className="px-4 py-3 text-center bg-blue-50 text-blue-600">盤點</th><th className="px-4 py-3 text-center">差異</th></tr></thead>
                    <tbody className="divide-y divide-gray-100">
                      {generateDiffList().map((item, idx) => (
                        <tr key={idx} className="hover:bg-gray-50">
                          <td className="px-4 py-3 flex items-center gap-3"><img src={item.image} className="w-8 h-8 rounded-md object-cover" /><div className="font-bold text-sm line-clamp-1">{item.name}</div></td>
                          <td className="px-4 py-3 text-center font-black">{item.size}</td><td className="px-4 py-3 text-center text-gray-500">{item.systemCount}</td><td className="px-4 py-3 text-center font-black text-blue-600 bg-blue-50/50">{item.scannedCount}</td>
                          <td className="px-4 py-3 text-center font-bold">{item.difference === 0 ? <span className="text-green-500">無</span> : item.difference > 0 ? <span className="text-blue-500">+{item.difference}</span> : <span className="text-red-500">{item.difference}</span>}</td>
                        </tr>
                      ))}
                    </tbody>
                  </table>
                </div>
                <div className="p-6 border-t border-gray-100 bg-gray-50 rounded-b-3xl flex gap-3">
                  <button onClick={() => { if(window.confirm("放棄盤點資料？")) { setScanSession({}); setScanLogs([]); setShowCompareModal(false); } }} className="px-6 py-4 rounded-xl font-bold text-gray-600 bg-white border border-gray-200">放棄</button>
                  <button onClick={confirmAndSaveStockTake} className="flex-1 bg-gray-900 text-white py-4 rounded-xl font-black text-lg">確認覆蓋系統數量</button>
                </div>
              </div>
            </div>
          )}

          {/* 手機底部導覽列 */}
          <nav className="md:hidden fixed bottom-0 left-0 right-0 bg-white/90 backdrop-blur-md border-t border-gray-100 pb-safe z-10 flex justify-around items-end px-2">
            <button onClick={() => setActiveTab('inventory')} className={`flex-1 pb-2 flex flex-col items-center gap-1 ${activeTab === 'inventory' ? 'text-gray-900' : 'text-gray-400'}`}><span className="text-xl">📦</span><span className="text-[10px] font-bold">總覽</span></button>
            <button onClick={() => setActiveTab('scan')} className={`flex-1 pb-2 flex flex-col items-center gap-1 ${activeTab === 'scan' ? 'text-gray-900' : 'text-gray-400'}`}><span className="text-xl">🔍</span><span className="text-[10px] font-bold">盤點</span></button>
            <button onClick={() => setActiveTab('upload')} className="relative flex flex-col items-center justify-center -top-4 w-[72px]">
              <div className={`w-14 h-14 rounded-full flex items-center justify-center text-white shadow-xl text-2xl ${activeTab === 'upload' ? 'bg-gray-900 shadow-gray-300' : 'bg-gray-800'}`}>➕</div><span className={`text-[10px] font-bold mt-1.5 ${activeTab === 'upload' ? 'text-gray-900' : 'text-gray-600'}`}>建檔</span>
            </button>
          </nav>
        </div>
      );
    }

    const root = ReactDOM.createRoot(document.getElementById('root'));
    root.render(<App />);
  </script>
